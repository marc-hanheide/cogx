# Start feature learning project.
PROJECT(HandPointer)

SET(SOURCES
  HandPointer.cpp HandDetector.cpp
)
SET(HEADERS
  HandPointer.h HandDetector.h
)

ADD_LIBRARY(${PROJECT_NAME} SHARED ${SOURCES})
ADD_EXECUTABLE(ColorLearnUtility ColorLearnUtility.cpp)

TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${OMNIORB4_LIBRARIES})
TARGET_LINK_LIBRARIES(${PROJECT_NAME} VisionIDL)
TARGET_LINK_LIBRARIES(${PROJECT_NAME} Vision VisionUtils)

# Use pkg-config to find opencv libraries...
INCLUDE(UsePkgConfig)
PKGCONFIG("opencv" OCV_INCLUDE_DIR OCV_LINK_DIR OCV_LINK_FLAGS OCV_CFLAGS)
TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${OCV_LINK_FLAGS})

INCLUDE_DIRECTORIES(${OCV_INCLUDE_DIR})
ADD_EXECUTABLE(ColorLearnUtility ColorLearnUtility.cpp)
TARGET_LINK_LIBRARIES(ColorLearnUtility ${OCV_LINK_FLAGS})

INSTALL(TARGETS ${PROJECT_NAME} LIBRARY DESTINATION lib)
INSTALL(FILES ${HEADERS} DESTINATION include/vision/components/HandPointer)
INSTALL(TARGETS ColorLearnUtility DESTINATION bin)
