PROJECT(VisualWorkingMemory)

SET(SOURCES VisualWorkingMemory.cpp)
SET(HEADERS VisualWorkingMemory.h)

ADD_LIBRARY(${PROJECT_NAME} SHARED ${SOURCES})

SET(OLD_CMAKE_FIND_LIBRARY_SUFFIXES ${CMAKE_FIND_LIBRARY_SUFFIXES})
SET(CMAKE_FIND_LIBRARY_SUFFIXES ".so" ".a")
FIND_PACKAGE(OpenGL)
INCLUDE_DIRECTORIES (${OPENGL_INCLUDE_DIR})
TARGET_LINK_LIBRARIES (${PROJECT_NAME} ${OPENGL_LIBRARIES})
FIND_PACKAGE(GLUT)
INCLUDE_DIRECTORIES (${GLUT_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${GLUT_LIBRARIES})
SET(CMAKE_FIND_LIBRARY_SUFFIXES ${OLD_CMAKE_FIND_LIBRARY_SUFFIXES})

TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${OMNIORB4_LIBRARIES})
TARGET_LINK_LIBRARIES(${PROJECT_NAME} FrameworkInterface FrameworkIDL FrameworkCore)
TARGET_LINK_LIBRARIES(${PROJECT_NAME} CASTCore CASTArchitecture CASTIDL)
TARGET_LINK_LIBRARIES(${PROJECT_NAME} Vision VisionUtils VisionIDL CoSyCommonIDLs)

INSTALL(TARGETS VisualWorkingMemory LIBRARY DESTINATION lib)
INSTALL(FILES ${HEADERS} DESTINATION include/vision/components/VisualWorkingMemory)
