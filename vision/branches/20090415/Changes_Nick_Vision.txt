
Copy to Nicks tar.gz:
-------------------------------------------
$CAST_ROOT/tools/math
$CAST_ROOT/tools/hardware
$CAST_ROOT/subarchitectures/vision.sa


Add the following to $CAST_ROOT/CMakeList.txt:
-------------------------------------------
# vision

add_subdirectory (tools/math)

option(BUILD_HAL_VIDEO "Build hardware video" YES)
if(BUILD_HAL_VIDEO)
add_subdirectory (tools/hardware/video)
endif(BUILD_HAL_VIDEO)

option(BUILD_SA_VISION "Build vision subarchitecure" YES)
if(BUILD_SA_VISION)
add_subdirectory (subarchitectures/vision.sa)
endif(BUILD_SA_VISION)


Define COGX_ROOT in $CAST_ROOT/CmakeLists.txt
---------------------------------------------
insert to # get current directory:
get_filename_component(COGX_ROOT . ABSOLUTE)


Include FindCAST.cmake:
---------------------------------------------
Add the following to # use cmake files packaged with CAST: in $CAST_ROOT/CMakeLists.txt

include(${CAST_ROOT}/cmake/FindCAST.cmake)


Define include pathes of cast headers:
-------------------------------------------
In $CAST_ROOT/CMakeLists.txt before # vision add:

include_directories(${CAST_INCLUDE_DIR}/cast/core)
include_directories(${CAST_INCLUDE_DIR}/cast/cdl)
include_directories(${CAST_INCLUDE_DIR}/cast/architecture)
include_directories(${CAST_INCLUDE_DIR}/cast/server)


Set LIBRARY_OUTPUT_PATH:
-------------------------------------------
cd $CAST_ROOT/BUILD
ccmake ..

set library output path to $CAST_ROOT/output/lib
e.g.: -> LIBRARY_OUTPUT_PATH = [homedir]/cogx/code/vision/branches/output/lib


Define direcotry of library (add to your .bashrc)
-------------------------------------------
export $LD_LIBRARY_PATH=/usr/local/lib:$LD_LIBRARY_PATH
